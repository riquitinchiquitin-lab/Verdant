FROM node:20-slim

WORKDIR /app

# Install dependencies for SQLite and healthcheck
RUN apt-get update && apt-get install -y python3 build-essential curl && rm -rf /var/lib/apt/lists/*

COPY package*.json ./
RUN npm install

COPY . .

# Build frontend during image creation, not container start
RUN npm run build

# Setup data directory
RUN mkdir -p /app/data && chmod 777 /app/data

EXPOSE 3000
ENV NODE_ENV=production

# Use node directly for faster startup
CMD ["node", "--loader", "ts-node/esm", "server.ts"]
